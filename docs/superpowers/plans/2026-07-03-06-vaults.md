# Sub-project 06 — Vaults Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Introduce vaults as a first-class concept — a `vaults` table, a `vault_id` FK on `tasks`, a full `tip vault` command surface, single-task `task move`, and active-vault selection wired through the storage handle.

**Architecture:** One `tip.db` holds a `vaults` table; every task carries a `vault_id NOT NULL` FK. The 03 `Vault` handle is renamed to `Store`, which resolves an active vault at open time (from `--vault` or `config.default_vault`). `store.tasks.*` is auto-scoped to the active vault; a new `store.vaults.*` handle does vault-registry CRUD (add/list/rename/merge/delete). `vault switch` reuses the 05 config machinery.

**Tech Stack:** Zig 0.16 (`std.Io`), `zig-sqlite`, `flags` dependency, `std.zon` (config).

**Dependency:** This plan requires **sub-projects 01–05 to be implemented first** — it relies on the error taxonomy (`src/core/errors.zig`, 01), the SQLite foundation + migration runner (`src/internal/database/db.zig`, `migrate.zig`, 02), the `Vault`→`Store` handle + `tasks` table (`src/core/vault.zig`, 03), the `complete`/`start` + unified prefix-matcher (04), and the config system + global flags (`src/core/config.zig`, `--vault` flag, 05).

## Global Constraints

- **Zig version:** 0.16.0 (`minimum_zig_version = "0.16.0"`). Use the new `std.Io` APIs.
- **Identifier casing:** functions/vars/fields = `snake_case`; types = `PascalCase`; enum members = `snake_case`; booleans use affirmative `is_`/`has_`/`can_`/`should_` prefix (code only, not CLI flags).
- **Storage:** one `tip.db`; vaults are rows; `tasks.vault_id TEXT NOT NULL REFERENCES vaults(id)`.
- **IDs:** ULID (sub-project 01), stored as TEXT, generated via `generate.generate_id`.
- **FK enforcement:** `PRAGMA foreign_keys = ON` on every connection open.
- **Default vault:** name `personal`, fixed bootstrap id `00000000000000000000000000` (valid 26-char Crockford ULID), seeded by migration `003`.
- **Precedence (active vault):** `--vault` flag > `config.default_vault` > `"personal"`.
- **Create verb:** `vault add` (supersedes charter D10's `vault init`).
- **Handle methods return data; CLI owns rendering** (03 S6).
- **Tests:** `zig build test --summary all` from repo root; storage tests use in-memory sqlite (`.mode = .{ .Memory = {} }`).
- **Error taxonomy (01 + this plan):** `TaskNotFound`, `AmbiguousPrefix`, `StorageFailure`, `EmptyTitle`, `VaultNotFound`, `VaultNameExists`, `VaultNotEmpty`, `CannotDeleteLastVault`, `CannotDeleteActiveVault`.
- **No global state.** Config/active-vault resolved in `main.zig` and passed explicitly.
- **Out of scope:** vault metadata beyond name (`description`/`color`/`icon`); per-vault encryption (10/11); `state.zon`; export/import (07); remote/sync (17+).

- [ ] **Step 1: Write the migration SQL**

Create `src/internal/database/migrations/003_create_vaults.sql`:

```sql
-- Sub-project 06: vaults table + tasks.vault_id FK.
-- The default vault uses a fixed bootstrap ULID so the migration stays a static
-- file (no runtime bind params) and tests are deterministic.

CREATE TABLE IF NOT EXISTS vaults (
    id         TEXT PRIMARY KEY NOT NULL,
    name       TEXT NOT NULL UNIQUE,
    created_at INTEGER NOT NULL
);

INSERT OR IGNORE INTO vaults (id, name, created_at)
VALUES ('00000000000000000000000000', 'personal', CAST(strftime('%s','now') AS INTEGER));

-- SQLite cannot add a NOT NULL FK column to a populated table in one ALTER,
-- so rebuild tasks.
CREATE TABLE tasks_new (
    id           TEXT PRIMARY KEY NOT NULL,
    vault_id     TEXT NOT NULL REFERENCES vaults(id),
    title        TEXT NOT NULL,
    description  TEXT,
    status       TEXT NOT NULL DEFAULT 'pending',
    priority     TEXT,
    due_date     INTEGER,
    assigned_to  TEXT,
    created_at   INTEGER NOT NULL,
    updated_at   INTEGER,
    completed_at INTEGER
);

INSERT INTO tasks_new
SELECT id, '00000000000000000000000000', title, description, status, priority,
       due_date, assigned_to, created_at, updated_at, completed_at
FROM tasks;

DROP TABLE tasks;
ALTER TABLE tasks_new RENAME TO tasks;

CREATE INDEX IF NOT EXISTS idx_tasks_status ON tasks(status);
CREATE INDEX IF NOT EXISTS idx_tasks_vault ON tasks(vault_id);

INSERT OR IGNORE INTO _schema_version (version) VALUES (3);
```

```zig
pub const OpenOptions = struct {
    vault: ?[]const u8 = null,
    default_vault: []const u8 = "personal",

pub fn open(allocator: std.mem.Allocator, io: std.Io, environ: std.process.Environ, opts: OpenOptions) !Store {
    var dir = try storage.dir.open_data_dir(allocator, io, environ);
    defer dir.close(io);
    const path = try dir.realpathAlloc(allocator, "tip.db"); // or join dir + "tip.db"
    defer allocator.free(path);

    var db = try database.open(allocator, path);
    errdefer db.deinit();
    try migrate.run_migrations(&db);
    return try init_from_db(allocator, io, db, opts);
}

fn init_from_db(allocator: std.mem.Allocator, io: std.Io, db: sqlite.Db, opts: OpenOptions) !Store {
    var store = Store{
        .db = db,
        .io = io,
        .active_vault_id = undefined,
        .tasks = undefined,
        .vaults = undefined,
    };
    store.tasks = .{ .store = &store };   // NOTE: re-point after return move (see Step 4 caveat)
    store.vaults = .{ .store = &store };

    const wanted = opts.vault orelse opts.default_vault;
    const v = store.vaults.get_by_name(allocator, wanted) catch |e| switch (e) {
        error.VaultNotFound => return error.VaultNotFound,
        else => return e,
    };
    defer allocator.free(v.id);
    store.active_vault_id = try allocator.dupe(u8, v.id);
    return store;
}

pub fn close(self: *Store) void {
    self.db.deinit();
}
```

```zig
pub fn add(self: *Vaults, allocator: std.mem.Allocator, name: []const u8) !models.Vault {
    const id = try generate.generate_id(allocator, self.store.io);
    errdefer allocator.free(id);
    const created_at = now_seconds(self.store.io);
    self.store.db.exec(
        "INSERT INTO vaults (id, name, created_at) VALUES (?, ?, ?)",
        .{}, .{ .id = id, .name = name, .created_at = created_at },
    ) catch |e| return map_unique(e); // UNIQUE(name) -> VaultNameExists
    return .{ .id = id, .name = name, .created_at = created_at };
}

pub fn get_by_name(self: *Vaults, allocator: std.mem.Allocator, name: []const u8) !models.Vault {
    const row = try self.store.db.oneAlloc(struct { id: []const u8, name: []const u8, created_at: i64 },
        allocator, "SELECT id, name, created_at FROM vaults WHERE name = ?", .{}, .{ .name = name });
    const r = row orelse return error.VaultNotFound;
    return .{ .id = r.id, .name = r.name, .created_at = r.created_at };
}

pub fn list(self: *Vaults, allocator: std.mem.Allocator) ![]models.Vault {
    var stmt = try self.store.db.prepare("SELECT id, name, created_at FROM vaults ORDER BY name");
    defer stmt.deinit();
    const rows = try stmt.all(struct { id: []const u8, name: []const u8, created_at: i64 }, allocator, .{}, .{});
    defer allocator.free(rows);
    var out = try allocator.alloc(models.Vault, rows.len);
    for (rows, 0..) |r, i| out[i] = .{ .id = r.id, .name = r.name, .created_at = r.created_at };
    return out;
}

pub fn count_tasks(self: *Vaults, vault_id: []const u8) !usize {
    const c = (try self.store.db.one(i64, "SELECT COUNT(*) FROM tasks WHERE vault_id = ?", .{}, .{ .vault_id = vault_id })) orelse 0;
    return @intCast(c);
}
```

```zig
pub fn rename(self: *Vaults, old_name: []const u8, new_name: []const u8) !void {
    // ensure target exists (loud) and new name is free
    self.store.db.exec("UPDATE vaults SET name = ? WHERE name = ?", .{},
        .{ .new_name = new_name, .old_name = old_name }) catch |e| return map_unique(e);
    const changed = self.store.db.rowsAffected();
    if (changed == 0) return error.VaultNotFound;
}

pub fn merge(self: *Vaults, allocator: std.mem.Allocator, src_name: []const u8, dst_name: []const u8, delete_src: bool) !void {
    const src = try self.get_by_name(allocator, src_name);
    defer { allocator.free(src.id); allocator.free(src.name); }
    const dst = try self.get_by_name(allocator, dst_name);
    defer { allocator.free(dst.id); allocator.free(dst.name); }

    try self.store.db.exec("UPDATE tasks SET vault_id = ? WHERE vault_id = ?", .{},
        .{ .dst = dst.id, .src = src.id });

    if (delete_src) {
        if (std.mem.eql(u8, src.id, self.store.active_vault_id)) return error.CannotDeleteActiveVault;
        try self.delete_row(src.id);
    }
}

pub fn delete(self: *Vaults, allocator: std.mem.Allocator, name: []const u8, force: bool) !void {
    const v = try self.get_by_name(allocator, name);
    defer { allocator.free(v.id); allocator.free(v.name); }

    if (std.mem.eql(u8, v.id, self.store.active_vault_id)) return error.CannotDeleteActiveVault;

    const total = (try self.store.db.one(i64, "SELECT COUNT(*) FROM vaults", .{}, .{})) orelse 0;
    if (total <= 1) return error.CannotDeleteLastVault;

    const task_count = try self.count_tasks(v.id);
    if (task_count > 0 and !force) return error.VaultNotEmpty;

    if (force and task_count > 0) {
        try self.store.db.exec("DELETE FROM tasks WHERE vault_id = ?", .{}, .{ .vault_id = v.id });
    }
    try self.delete_row(v.id);
}

fn delete_row(self: *Vaults, id: []const u8) !void {
    try self.store.db.exec("DELETE FROM vaults WHERE id = ?", .{}, .{ .id = id });
}
```

> Ordering note in `delete`: check active-vault first, then last-vault, then emptiness — matches the test expectations (deleting active `personal` yields `CannotDeleteActiveVault`, not `CannotDeleteLastVault`).

```zig
fn switch_vault(allocator: std.mem.Allocator, io: std.Io, environ: std.process.Environ, name: []const u8) !void {

}
```
